```shellscript cURL
curl -X POST "https://decode.abingdoncs.org.uk/v1/chat/completions" \
  -H "Authorization: Bearer " \
  -H "Content-Type: application/json" \
  -d '{
    "model": "gpt-4o",
    "messages": [
      {
        "role": "system",
        "content": "string"
      }
    ]
  }'
```


```python Python
import requests

url = "https://decode.abingdoncs.org.uk/v1/chat/completions"
body = """{
  "model": "gpt-4o",
  "messages": [
    {
      "role": "system",
      "content": "string"
    }
  ]
}"""
response = requests.request("POST", url, data = body, headers = {
  "Content-Type": "application/json", 
  "Authorization": "Bearer "
})

print(response
```
